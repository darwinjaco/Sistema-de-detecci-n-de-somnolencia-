#include <WiFi.h>
#include <HTTPClient.h>
#include <WiFiClientSecure.h>
#include <BluetoothSerial.h>
#include <EEPROM.h>
#include <WebServer.h>

// CONFIGURACION TELEGRAM
const char* botToken = "8053729564:AAFElZRBjO04zOI1qj2ur6BOOShEUFV9Oao";
const char* chatId = "1168725927";

// PINES DE SALIDA
//const int pinLED = 21; // LED YA NO SE USA
//const int pinBuzzer = 19; // BUZZER YA NO SE USA
const int pinMotor = 18;

// CONFIGURACION PWM
const int pwmChannel = 0;
const int freq = 5000;
const int resolution = 8;

// CONFIGURACION EEPROM
#define EEPROM_SIZE 200
#define SSID1_ADDR 0
#define PASS1_ADDR 50
#define SSID2_ADDR 100
#define PASS2_ADDR 150

// WIFI DE ESP32
WebServer server(80);
const char* ap_ssid = "ESP32_Config";
const char* ap_password = "12345678";

BluetoothSerial SerialBT;

// FUNCION ESCRIBIR EN EEPROM
void writeStringToEEPROM(int addr, const String &data) {
  for (int i = 0; i < 50; ++i)
    EEPROM.write(addr + i, i < data.length() ? data[i] : 0);
}

// FUNCION LEER DE EEPROM
String readStringFromEEPROM(int addr) {
  char data[51];
  for (int i = 0; i < 50; ++i)
    data[i] = EEPROM.read(addr + i);
  data[50] = '\0';
  return String(data);
}

// ESPERAR MIENTRAS CONECTA EL WIFI
bool waitForConnection() {
  for (int i = 0; i < 10; i++) {
    if (WiFi.status() == WL_CONNECTED) {
      Serial.print("Conectado a: ");
      Serial.println(WiFi.localIP());
      return true;
    }
    delay(500);
  }
  return false;
}

// CONEXION CON RED WIFI1 O WIFI2
bool connectToStoredNetworks() {
  String ssid1 = readStringFromEEPROM(SSID1_ADDR);
  String pass1 = readStringFromEEPROM(PASS1_ADDR);
  String ssid2 = readStringFromEEPROM(SSID2_ADDR);
  String pass2 = readStringFromEEPROM(PASS2_ADDR);

  if (ssid1.length() > 0) {
    WiFi.begin(ssid1.c_str(), pass1.c_str());
    Serial.print("Intentando conectar a: "); Serial.println(ssid1);
    if (waitForConnection()) return true;
  }
  if (ssid2.length() > 0) {
    WiFi.begin(ssid2.c_str(), pass2.c_str());
    Serial.print("Intentando conectar a: "); Serial.println(ssid2);
    if (waitForConnection()) return true;
  }
  return false;
}

void handleSave() {
  writeStringToEEPROM(SSID1_ADDR, server.arg("ssid1"));
  writeStringToEEPROM(PASS1_ADDR, server.arg("pass1"));
  writeStringToEEPROM(SSID2_ADDR, server.arg("ssid2"));
  writeStringToEEPROM(PASS2_ADDR, server.arg("pass2"));
  EEPROM.commit();
  server.send(200, "text/html", "<h2>Redes guardadas ✅<br>Reinicia la ESP32-CAM</h2>");
}

void setupAPWebServer() {
  server.on("/", handleRoot);
  server.on("/save", HTTP_POST, handleSave);
  server.begin();
}

// FUNCION PARA ENVIAR MENSAJE POR TELEGRAM (OPTIMIZADA)
void enviarTelegram(const String& mensaje) {
  if (WiFi.status() == WL_CONNECTED) {
    SerialBT.end();   // 🔥 Desactiva Bluetooth para liberar RAM
    delay(300);

    WiFiClientSecure client;
    client.setInsecure();

    HTTPClient http;
    char url[512];
    snprintf(url, sizeof(url),
      "https://api.telegram.org/bot%s/sendMessage?chat_id=%s&text=%s",
      botToken, chatId, mensaje.c_str());

    if (http.begin(client, url)) {
      int httpCode = http.GET();
      if (httpCode > 0) {
        Serial.println("✅ Mensaje enviado por Telegram.");
      } else {
        Serial.printf("❌ Error HTTP: %d\n", httpCode);
      }
      http.end();
    } else {
      Serial.println("❌ No se pudo iniciar la conexión HTTPS.");
    }

    delay(300); // Opcional para estabilidad
    SerialBT.begin("ESP32-Somnolencia");  // 🔥 Reactiva Bluetooth
  } else {
    Serial.println("❌ No hay conexión WiFi.");
  }
}

// FUNCIONES POR FASES DE SOMNOLENCIA
void fase1() {
  Serial.println("Fase 1: Somnolencia leve (sin motor)");
  ledcWrite(pwmChannel, 0); // Motor apagado
}

void fase2() {
  Serial.println("Fase 2: Somnolencia media (motor 50%)");
  ledcWrite(pwmChannel, 128);  // 50%
  enviarTelegram("⚠️ Fase 2: Somnolencia intermedia detectada.");
  delay(2000); // Motor 2s activo
  ledcWrite(pwmChannel, 0);
}

void fase3() {
  Serial.println("Fase 3: Somnolencia crítica (motor 100%)");
  ledcWrite(pwmChannel, 255);  // Motor al 100%
  enviarTelegram("🚨 ¡Fase 3: Nivel crítico de somnolencia detectado!");
  delay(2000); // Motor 2s activo
  ledcWrite(pwmChannel, 0);  // Apagar motor
}

//SETUP---------------------------------------------------------
void setup() {
  Serial.begin(115200);
  //pinMode(pinLED, OUTPUT);   // LED desactivado
  //pinMode(pinBuzzer, OUTPUT); // Buzzer desactivado
  ledcSetup(pwmChannel, freq, resolution);
  ledcAttachPin(pinMotor, pwmChannel);

  EEPROM.begin(EEPROM_SIZE);

  // Solo inicializa el WebServer si falla la conexión WiFi
  if (!connectToStoredNetworks()) {
    Serial.println("No se conectó. Iniciando AP WiFi...");
    WiFi.mode(WIFI_AP);
    WiFi.softAP(ap_ssid, ap_password);
    Serial.print("AP Activo. IP: ");
    Serial.println(WiFi.softAPIP());
    setupAPWebServer();
  } else {
    Serial.println("WiFi conectado correctamente.");
  }

  SerialBT.begin("ESP32-Somnolencia");
  Serial.println("Bluetooth listo. Esperando comandos...");
}

//LOOP----------------------------------------------------------
void loop() {
  // Solo procesa el WebServer si está activo (modo AP)
  if (WiFi.getMode() == WIFI_AP) {
    server.handleClient();
  }

  if (SerialBT.available()) {
    char command = SerialBT.read();
    if (command == '1' || command == '2' || command == '3') {
      Serial.print("Comando recibido: somnolencia_");
      Serial.println(command);
      switch (command) {
        case '1': fase1(); break;
        case '2': fase2(); break;
        case '3': fase3(); break;
      }
    } else {
      Serial.print("Comando inválido recibido: ");
      Serial.println(command);
    }
  }

  delay(10);
}
